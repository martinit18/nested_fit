% It is written in tex because it is easier to document / incorporate with other functions

heaviside(x)                    = 1 + \texttt{sign}(x)
gauss(x, x_0, A_0, s)           = \frac{A_0}{\sqrt{2\pi}s}\exp{-\frac{(x-x_0)^2}{2s^2}}
supergauss(x, x_0, A_0, s)      = \frac{A_0\sqrt(2.0)}{s\Gamma(0.25)}\exp{-\frac{(x-x_0)^4}{4s^4}}
erfpeak(x, x_0, A_0, s, w)      = \frac{A_0}{4w}(\texttt{erf}(\frac{w + x - x_0}{\sqrt{2.0}s}) + \texttt{erf}(\frac{w - x + x_0}{\sqrt{2.0}s}))
lorentzian(x, x_0, A_0, g)      = \frac{A_0 g}{\pi((x-x_0)^2+g^2)}
lorentziannrm(x, x_0, A_0, g)   = \frac{A_0 g}{2\pi}\frac{1}{(x-x_0)^2 + g^2/4}
voigt(x, x_0, A_0, s, g)        = \frac{A_0 \texttt{WofzRe}(\frac{x-x_0}{s\sqrt{2.0}}, \frac{g}{s\sqrt{2.0}})}{s\sqrt{2.0\pi}}
